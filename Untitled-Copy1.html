Big Mart Sales Prediction¶

数据来源:https://datahack.analyticsvidhya.com/contest/practice-problem-big-mart-sales-iii/

分析目的：¶

通过给定信息 探索影响销售额的因素 预测特定商品在特定商店的销售额。

#以相同产品在不同类型商店可见度的平均值来填充

for i,row in all_data.loc[all_data['Item_Visibility']==0,:].iterrows():
    s = row['Item_Identifier']
    g = row['Outlet_Type']
    if(g == "Grocery Store"):
        #在多个 Grocery 中销售的同一个商品，在其中一个商店的值缺失  用其他Grocery 中的平均值填充
        all_data.loc[i,'Item_Visibility'] = all_data[(all_data.Item_Identifier == s)&(all_data.Outlet_Type =='Grocery Store')&(all_data.Item_Visibility != 0)].Item_Visibility.mean()
        
        
    elif g != "Grocery Store":
        #在多个 非 Grocery 中销售的同一个商品  用 非Grocery 的平均值填充
        all_data.loc[i,'Item_Visibility'] = all_data[(all_data.Item_Identifier == s)&(all_data.Outlet_Type !='Grocery Store')&(all_data.Item_Visibility != 0)].Item_Visibility.mean()
In [17]:
#如果某商品 在 Grocery 中的值全部缺失  上面的处理会引入nan 那么需要用 Supmarket 的值来填充 从前面的数据可以看到 Grocery 中的值是Supermarket 的1.69倍

for i,row in all_data.loc[all_data.Item_Visibility.isnull(),:].iterrows():
    s = row['Item_Identifier']
    g = row['Outlet_Type']
    if g == 'Grocery Store' :
        # Grocery 中的数据全部缺失，用Supmarket 的平均值的1.69倍填充
        all_data.Item_Visibility.fillna((all_data[(all_data.Item_Identifier == s)&(all_data.Item_Visibility != 0)].Item_Visibility.mean()*1.69),inplace = True)
    if g != 'Grocery Store':
        # Supermarket 中的数据全部缺失 用Grocery 的平均值除以1.69 填充
            all_data.Item_Visibility.fillna((all_data[(all_data.Item_Identifier == s)&(all_data.Item_Visibility != 0)].Item_Visibility.mean()/1.69),inplace = True)

import re
def _parse_ID(s):
    s = s.strip()
    
    try:
        p1 = re.compile("[A-Z]+")
        p2 = re.compile("[\d]+")
        
        return p1.findall(s)[0],p2.findall(s)[0]
    except:
        return np.nan,np.nan
