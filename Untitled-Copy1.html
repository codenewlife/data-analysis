import re
def _parse_ID(s):
    s = s.strip()
    
    try:
        p1 = re.compile("[A-Z]+")
        p2 = re.compile("[\d]+")
        
        return p1.findall(s)[0],p2.findall(s)[0]
    except:
        return np.nan,np.nan
